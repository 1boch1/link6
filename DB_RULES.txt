rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Regole per la collezione "users"
    match /users/{userId} {
      // Permetti a chiunque sia autenticato di leggere tutti gli utenti
      allow read: if request.auth != null;
      // Permetti la scrittura solo al proprietario del documento
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Regole per la collezione "requests"
    match /requests/{requestId} {
      // Un utente può leggere solo le richieste che ha creato o quelle create dai suoi amici
      allow read: if request.auth != null && (
        request.auth.token.email == resource.data.createdBy ||
        isFriend(request.auth.token.email, resource.data.createdBy)
      );
      
      allow create, update: if request.auth != null;
     
    }

    // Regole per la collezione "chats"
    match /chats/{chatId} {
      // Solo gli utenti coinvolti nella chat possono leggerla o scriverci
      allow read, write: if request.auth != null && request.auth.token.email in resource.data.users;
    }
    
    // Funzione per controllare se un utente è amico di un altro
    function isFriend(userEmail, friendEmail) {
      return exists(/databases/$(database)/documents/users/$(userEmail)) &&
             friendEmail in get(/databases/$(database)/documents/users/$(userEmail)).data.amici;
    }
  }
}
